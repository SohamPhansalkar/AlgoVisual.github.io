      #include <stdio.h>
      int main(){
          int arr[] = {64, 34, 25, 12, 22, 11, 90};
          int n = sizeof(arr)/sizeof(arr[0]);
          int i, j, temp;
          for (i = 0; i < n-1; i++)
              for (j = 0; j < n-i-1; j++)
                  if (arr[j] > arr[j+1]) {
                      temp = arr[j];
                      arr[j] = arr[j+1];
                      arr[j+1] = temp;
                  }
                  
          printf("Sorted array: \n");
          for (i=0; i < n; i++)
              printf("%d ", arr[i]);
          return 0;
      }
